#+title: M1 Functions

* Базовые понятия функции :asm:
** Определение
| Инструкции для вызова функций                                  | x86      | ARM64  | RISC-V          |
| Вызов процедуры, располагающейся по фиксированной метке/адресу | call     | bl     | jal ra, label   |
| Возврат из процедуры                                           | ret      | ret    | jalr x0, ra, 0  |
| Вызов процедуры по адресу, хранимому в регистре                | call eax | blr x0 | jalr ra, reg, 0 |

Функцию опеределяет ее вызов - переход по определенному адресу, перед вызовом должны быть подготовлены аргументы, а после вызова выполняется переход на предыдущий адрес возврата
- Фрейм - область стека в которой содержится информация для определенной функции
  Аргументы, локальные переменные, адрес возврата, пролог, эпилог
  Пролог сохраняет значения регистров используемых внутри функции, записывает в стек аргументы
  Эпилог восстанавливает значения регистров сохраненные прологом
  Очищает стек

  | higher         |
  | arg 1          |
  | ...            |
  | arg n          |
  | return address |
  | previous BP    |
  | local var 1    |
  | ...            |
  | local var n    |
  | lower          |
   Допустим rsp ниже
   rbp на previous rbp
   pop rbp := rbp = [rsp], rsp += 8
   rsp стал previous BP
   ret := rip = [rsp], rsp +=8

** Пример x86
#+begin_src c
int dummy() {
    int a = 5;
    int b = 4;
    int c = a + b;
    return c;
}
#+end_src

#+begin_src asm
; Пролог
push    rbp
mov     rbp, rsp

; Инициализация переменных
mov     DWORD PTR [rbp-4], 5
mov     DWORD PTR [rbp-8], 4

; Тело функции
mov     edx, DWORD PTR [rbp-4]
mov     eax, DWORD PTR [rbp-8]
add     eax, edx
mov     DWORD PTR [rbp-12], eax

; Эпилог функции
mov     eax, DWORD PTR [rbp-12]
pop     rbp
ret
#+end_src

push: сохраняем rbp - base pointer который является указателем на базовый адрес фрейма вызывающей функции
rbp = rsp: обновляем rbp для фрейма текущей функции
rbp-n: смещения относительно rbp в которых хранятся локальные переменные
Логика функции происходит в теле
eax: регистр храняший результат функции
pop rbp ~ rbp = *rsp, rsp = rsp + 8: восстанвливаем старый rbp
ret ~ rip = *rsp, rsp = rsp+8

** Пример ARM
#+begin_src asm
sub     sp, sp, #16
        mov     w0, 5
        str     w0, [sp, 12]
        mov     w0, 4
        str     w0, [sp, 8]
        ldr     w1, [sp, 12]
        ldr     w0, [sp, 8]
        add     w0, w1, w0
        str     w0, [sp, 4]
        ldr     w0, [sp, 4]
        add     sp, sp, 16
        ret
#+end_src
